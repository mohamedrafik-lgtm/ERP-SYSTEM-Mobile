## 🛠️ دوال المساعدة (Helper Functions)

### A. Token Utilities
**المسار**: [`src/lib/trainee-api.ts`](src/lib/trainee-api.ts:1)

```typescript
// فحص إذا Token قريب من الانتهاء
export function isTokenExpiringSoon(token: string): boolean {
  const payload = JSON.parse(atob(token.split('.')[1]));
  const expiryTime = payload.exp * 1000;
  const now = Date.now();
  const timeLeft = expiryTime - now;
  
  // إذا باقي أقل من 30 دقيقة
  return timeLeft < 30 * 60 * 1000;
}

// معالجة انتهاء الصلاحية
export function handleTokenExpiry() {
  localStorage.removeItem('trainee_token');
  localStorage.removeItem('trainee_data');
  window.location.href = '/trainee-auth';
}

// فحص إذا الخطأ بسبب Token منتهي
export function isTokenExpiryError(error: any): boolean {
  return error.status === 401 || 
         error.message?.includes('Token expired') ||
         error.message?.includes('jwt expired');
}
```
