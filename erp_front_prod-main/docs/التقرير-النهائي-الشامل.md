**الملف الجديد:** `src/lib/batch-processor.ts`

```typescript
/**
 * معالجة مصفوفة على دفعات لمنع Memory/CPU Spikes
 */
export async function processBatchesWithFallback<T, R>(
  items: T[],
  processor: (item: T) => Promise<R>,
  fallbackValue: R,
  batchSize: number = 5
): Promise<R[]> {
  const results: R[] = [];
  
  // معالجة على دفعات
  for (let i = 0; i < items.length; i += batchSize) {
    const batch = items.slice(i, i + batchSize);
    
    // معالجة الدفعة الحالية
    const batchResults = await Promise.all(
      batch.map(async (item) => {
        try {
          return await processor(item);
        } catch (error) {
          return fallbackValue; // fallback عند الفشل
        }
      })
    );
    
    results.push(...batchResults);
  }
  
  return results;
}
```

**ثانياً: تطبيق في content/page.tsx**
